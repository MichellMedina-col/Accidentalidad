def capturar_datos_y_predecir():
    print("="*50)
    print("🚦 SISTEMA DE PREDICCIÓN DE RIESGO DE ACCIDENTE")
    print("="*50)
    
    try:
        cols_modelo = list(modelo.feature_names_in_)
        clases = modelo.classes_ # Esto nos dice qué significan los 4 números

        # Captura de datos
        edad = int(input("Edad: "))
        municipio = input("📍 Ingrese municipio (Funza, Madrid, Mosquera, Facatativá): ").title()
        zona = input("🌎 Ingrese zona (Urbana, Rural): ").title()
        genero = input("👤 Ingrese género (Masculino, Femenino): ").title()
        actor = input("🚗 Ingrese actor vial (Motociclista, Conductor, Peatón, Ciclista, Pasajero): ").title()

        # Crear entrada
        entrada = pd.DataFrame(0, index=[0], columns=cols_modelo)
        if 'Edad' in entrada.columns:
            entrada.at[0, 'Edad'] = edad

        # Mapeo exacto según tus columnas detectadas
        mapeo = {
            f'Municipio_{municipio}': 1,
            f'Zona_{zona}': 1,
            f'Genero_{genero}': 1,
            f'Actor_vial_{actor}': 1
        }

        for col, val in mapeo.items():
            if col in entrada.columns:
                entrada.at[0, col] = val

        # Predicción
        prediccion = modelo.predict(entrada)[0]
        probabilidades = modelo.predict_proba(entrada)[0]
        
        # Obtener la probabilidad más alta
        max_prob = max(probabilidades)

        print("\n" + "="*50)
        print(f"📊 RESULTADO DE PREDICCIÓN: Clase {prediccion}")
        print(f"Confianza del modelo: {max_prob:.2%}")
        print("-" * 50)
        print("Desglose de probabilidades por categoría:")
        for i, clase in enumerate(clases):
            print(f" - Categoría {clase}: {probabilidades[i]:.2%}")
        print("="*50)

    except Exception as e:
        print(f"⚠️ Error: {e}")

# Ejecutar
capturar_datos_y_predecir()
